//6. Write a function to print data of all students who joined in 2019. 

#include <stdio.h>

struct student{
    char Name[20];
    int age;
    float grade;
    int year;   // year of joining
};

// Function to print students who joined in 2019
void print2019Students(struct student s[], int n){
    printf("\nStudents who joined in 2019:\n");

    for(int i=0; i<n; i++){
        if(s[i].year == 2019){
            printf("\nName : %s\n", s[i].Name);
            printf("Age : %d\n", s[i].age);
            printf("Grade : %.2f\n", s[i].grade);
            printf("Year : %d\n", s[i].year);
        }
    }
}

int main(){

    struct student s[5];

    // Input
    for(int i=0; i<5; i++){
        printf("Enter Name of student %d: ", i+1);
        scanf("%s", s[i].Name);

        printf("Enter Age: ");
        scanf("%d", &s[i].age);

        printf("Enter Grade: ");
        scanf("%f", &s[i].grade);

        printf("Enter Year of Joining: ");
        scanf("%d", &s[i].year);
    }

    // Function call
    print2019Students(s, 5);

    return 0;
}
