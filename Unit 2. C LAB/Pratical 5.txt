//5. Write a function to print names of students whose grade is greater 
than 5.0
#include <stdio.h>

struct student{
    char Name[20];
    int age;
    float grade;
};

// Function to print students with grade > 5.0
void printTopStudents(struct student s[], int n){
    printf("\nStudents with grade greater than 5.0:\n");

    for(int i=0; i<n; i++){
        if(s[i].grade > 5.0){
            printf("%s\n", s[i].Name);
        }
    }
}

int main(){
    
    struct student s[5];

    // Input
    for(int i=0; i<5; i++){
        printf("Enter the Name of student %d: ", i+1);
        scanf("%s", s[i].Name);
        
        printf("Enter the Age of student %d: ", i+1);
        scanf("%d", &s[i].age);
        
        printf("Enter the grade of student %d: ", i+1);
        scanf("%f", &s[i].grade);
    }

    // Function call
    printTopStudents(s, 5);

    return 0;
}
