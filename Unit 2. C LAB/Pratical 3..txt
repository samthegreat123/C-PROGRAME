//3. Sort above data by name. 

#include <stdio.h>
#include <string.h>

struct student{
    char Name[20];
    int age;
    float grade;
};

int main(){
    
    struct student s[5], temp;

    // Input
    for(int i=0; i<5; i++){
        printf("Enter the Name of student %d: ", i+1);
        scanf("%s", s[i].Name);
        
        printf("Enter the Age of student %d: ", i+1);
        scanf("%d", &s[i].age);
        
        printf("Enter the grade of student %d: ", i+1);
        scanf("%f", &s[i].grade);
    }

    // Sorting (by name)
    for(int i=0; i<5; i++){
        for(int j=i+1; j<5; j++){
            if(strcmp(s[i].Name, s[j].Name) > 0){
                temp = s[i];
                s[i] = s[j];
                s[j] = temp;
            }
        }
    }

    // Output
    printf("\n---STUDENT RECORDS (SORTED BY NAME)----\n");

    for(int i=0; i<5; i++){
        printf("\nStudent %d\n", i+1);
        printf("Name : %s\n", s[i].Name);
        printf("Age : %d\n", s[i].age);
        printf("Grade : %.2f\n", s[i].grade);
    }

    return 0;
}
