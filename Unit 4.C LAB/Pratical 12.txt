//12. Wap to enter a number in one file and print its reverse in another 
file.
#include <stdio.h>

int main() {
    FILE *fp1, *fp2;
    int num, rev = 0, rem, temp;

    // Step 1: Write number into file
    fp1 = fopen("number.txt", "w");

    if (fp1 == NULL) {
        printf("Error opening file!\n");
        return 1;
    }

    printf("Enter a number: ");
    scanf("%d", &num);

    fprintf(fp1, "%d", num);
    fclose(fp1);

    // Step 2: Read number from file
    fp1 = fopen("number.txt", "r");
    fscanf(fp1, "%d", &temp);
    fclose(fp1);

    // Step 3: Reverse the number
    while (temp != 0) {
        rem = temp % 10;
        rev = rev * 10 + rem;
        temp = temp / 10;
    }

    // Step 4: Write reversed number into another file
    fp2 = fopen("reverse.txt", "w");

    if (fp2 == NULL) {
        printf("Error opening file!\n");
        return 1;
    }

    fprintf(fp2, "%d", rev);
    fclose(fp2);

    printf("Reversed number written to reverse.txt\n");

    return 0;
}
