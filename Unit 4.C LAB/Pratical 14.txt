14. Wap to copy one file into another using command line arguments.
#include <stdio.h>

int main(int argc, char *argv[]) {
    FILE *fp1, *fp2;
    char ch;

    // Check arguments
    if (argc != 3) {
        printf("Usage: program sourcefile destinationfile\n");
        return 1;
    }

    // Open source file
    fp1 = fopen(argv[1], "r");
    if (fp1 == NULL) {
        printf("Error opening source file!\n");
        return 1;
    }

    // Open destination file
    fp2 = fopen(argv[2], "w");
    if (fp2 == NULL) {
        printf("Error opening destination file!\n");
        fclose(fp1);
        return 1;
    }

    // Copy content
    while ((ch = fgetc(fp1)) != EOF) {
        fputc(ch, fp2);
    }

    fclose(fp1);
    fclose(fp2);

    printf("File copied successfully!\n");

    return 0;
}
OUTPUT:
File copied successfully!
