//11. Wap to find the size of file without traversing character by 
character. 
#include <stdio.h>

int main() {
    FILE *fp;
    long size;

    fp = fopen("sample.txt", "r");

    if (fp == NULL) {
        printf("File not found!\n");
        return 1;
    }

    // Move file pointer to end
    fseek(fp, 0, SEEK_END);

    // Get current position (size of file)
    size = ftell(fp);

    fclose(fp);

    printf("Size of file = %ld bytes\n", size);

    return 0;
}

OUTPUT:
Size of file = 39 bytes
