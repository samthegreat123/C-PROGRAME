//10. Wap to read a file and replace all ‘a’ with ‘x’.
#include <stdio.h>

int main() {
    FILE *fp1, *fp2;
    char ch;

    // Open source file
    fp1 = fopen("input.txt", "r");
    if (fp1 == NULL) {
        printf("Error opening input file!\n");
        return 1;
    }

    // Open destination file
    fp2 = fopen("output.txt", "w");
    if (fp2 == NULL) {
        printf("Error opening output file!\n");
        fclose(fp1);
        return 1;
    }

    // Read and replace
    while ((ch = fgetc(fp1)) != EOF) {
        if (ch == 'a')
            ch = 'x';

        fputc(ch, fp2);
    }

    printf("Replacement done successfully!\n");

    fclose(fp1);
    fclose(fp2);

    return 0;
}
OUTPUT:
Replacement done successfully!
