#include <stdio.h>

int main(){
    char s[100];
    int i=0, len=0, flag=0;

    printf("Enter a string: ");
    fgets(s, sizeof(s), stdin);

    while(s[len] != '\0' && s[len] != '\n')
        len++;

    for(i=0; i<len/2; i++){
        if(s[i] != s[len-1-i]){
            flag = 1;
            break;
        }
    }

    if(flag == 0)
        printf("Palindrome");
    else
        printf("Not Palindrome");

    return 0;
}
