#include <stdio.h>

int main(){
    char name[50];
    int i = 0;

    printf("Enter your name: ");
    fgets(name, sizeof(name), stdin);

    while(name[i] != '\0'){
        if(name[i] != '\n')   // avoid extra line
            printf("%c\n", name[i]);
        i++;
    }

    return 0;
}
