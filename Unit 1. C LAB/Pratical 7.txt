#include <stdio.h>

int main(){
    int n1, n2, i;

    printf("Enter size of first array: ");
    scanf("%d", &n1);

    int a[n1];
    printf("Enter %d elements:\n", n1);
    for(i=0; i<n1; i++)
        scanf("%d", &a[i]);

    printf("Enter size of second array: ");
    scanf("%d", &n2);

    int b[n2];
    printf("Enter %d elements:\n", n2);
    for(i=0; i<n2; i++)
        scanf("%d", &b[i]);

    int c[n1+n2];

    // Copy first array
    for(i=0; i<n1; i++)
        c[i] = a[i];

    // Copy second array
    for(i=0; i<n2; i++)
        c[n1+i] = b[i];

    printf("Merged array:\n");
    for(i=0; i<n1+n2; i++)
        printf("%d ", c[i]);

    return 0;
}
