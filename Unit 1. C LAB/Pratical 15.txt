#include <stdio.h>

int main(){
    char s[100];
    int i = 0, spaces = 0, vowels = 0;

    printf("Enter a string: ");
    fgets(s, sizeof(s), stdin);

    while(s[i] != '\0'){
        if(s[i] == ' ')
            spaces++;

        if(s[i]=='a'||s[i]=='e'||s[i]=='i'||s[i]=='o'||s[i]=='u'||
           s[i]=='A'||s[i]=='E'||s[i]=='I'||s[i]=='O'||s[i]=='U')
            vowels++;

        i++;
    }

    printf("Spaces = %d\n", spaces);
    printf("Vowels = %d", vowels);

    return 0;
}
