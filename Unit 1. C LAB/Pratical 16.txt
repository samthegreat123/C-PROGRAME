#include <stdio.h>
#include <string.h>

int main(){
    char s1[100], s2[100];

    printf("Enter first string: ");
    fgets(s1, sizeof(s1), stdin);

    printf("Enter second string: ");
    fgets(s2, sizeof(s2), stdin);

    if(strcmp(s1, s2) == 0)
        printf("Strings are same");
    else
        printf("Strings are different");

    return 0;
