//5. Enter N elements and find out sum of prime numbers using dynamic 
array. 
#include <stdio.h>
#include <stdlib.h>

// Function to check prime
int isPrime(int num){
    if(num <= 1)
        return 0;

    for(int i = 2; i <= num/2; i++){
        if(num % i == 0)
            return 0;
    }
    return 1;
}

int main(){

    int N, sum = 0;

    printf("Enter number of elements: ");
    scanf("%d", &N);

    // Dynamic array
    int *arr = (int*)malloc(N * sizeof(int));

    // Input elements
    for(int i = 0; i < N; i++){
        printf("Enter element %d: ", i+1);
        scanf("%d", &arr[i]);
    }

    // Find sum of prime numbers
    for(int i = 0; i < N; i++){
        if(isPrime(arr[i])){
            sum = sum + arr[i];
        }
    }

    printf("\nSum of prime numbers = %d\n", sum);

    // Free memory
    free(arr);

    return 0;
}
