//4. Enter N elements and find out sum and average of them using 
dynamic array. 
#include <stdio.h>
#include <stdlib.h>

int main(){

    int N;
    float sum = 0, average;

    printf("Enter number of elements: ");
    scanf("%d", &N);

    // Dynamic memory allocation
    int *arr = (int*)malloc(N * sizeof(int));

    // Input elements
    for(int i = 0; i < N; i++){
        printf("Enter element %d: ", i+1);
        scanf("%d", &arr[i]);
        sum = sum + arr[i];
    }

    // Calculate average
    average = sum / N;

    printf("\nSum = %.2f\n", sum);
    printf("Average = %.2f\n", average);

    // Free memory
    free(arr);

    return 0;
}
