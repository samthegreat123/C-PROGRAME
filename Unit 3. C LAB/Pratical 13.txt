// Enter a string and print it in reverse order. 
#include <stdio.h>
#include <string.h>

int main(){

    char str[50];
    int len;

    // Input
    printf("Enter a string: ");
    scanf("%s", str);

    // Find length
    len = strlen(str);

    // Reverse and print
    printf("Reversed string = ");
    for(int i = len - 1; i >= 0; i--){
        printf("%c", str[i]);
    }

    return 0;
}
OUTPUT:
Enter a string: hello
Reversed string = olleh
