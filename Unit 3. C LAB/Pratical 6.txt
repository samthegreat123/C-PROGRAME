//Enter N elements and find out sum of odd and even numbers using 
dynamic array 
#include <stdio.h>
#include <stdlib.h>

int main(){

    int N;
    int sum_even = 0, sum_odd = 0;

    printf("Enter number of elements: ");
    scanf("%d", &N);

    // Dynamic array allocation
    int *arr = (int*)malloc(N * sizeof(int));

    // Input elements
    for(int i = 0; i < N; i++){
        printf("Enter element %d: ", i+1);
        scanf("%d", &arr[i]);
    }

    // Calculate sum of even and odd
    for(int i = 0; i < N; i++){
        if(arr[i] % 2 == 0){
            sum_even += arr[i];
        } else {
            sum_odd += arr[i];
        }
    }

    // Output
    printf("\nSum of even numbers = %d\n", sum_even);
    printf("Sum of odd numbers = %d\n", sum_odd);

    // Free memory
    free(arr);

    return 0;
}

Enter element 5: 5

Sum of even numbers = 6
Sum of odd numbers = 9
